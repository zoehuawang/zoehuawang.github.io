一、常用方法 

#模拟发送option探测
sipsak -H 本地ip -l 本地端口 -s sip:192.168.7.55:18627 -v

#模拟发送invite请求，-f从文件加载sip消息，用 ! 分割文件中的变量和值
sipsak -f invite.txt -g '!CALLEE!被叫号码!CALL_ID!'$(uuidgen)'!CALLER!主叫号码!SIP_PORT!本地信令端口!MEDIA_PORT!本地媒体端口!' -H 192.168.7.51 -l 本地信令端口 -s sip:192.168.7.55:18627 -v

#文件invite.txt内容模版如下:
INVITE sip:$CALLEE$@192.168.7.55:18627 SIP/2.0
Call-ID: $CALL_ID$
From: "$CALLER$"<sip:$CALLER$@192.168.7.51;transport=udp;user=phone>
To: "$CALLEE$"<sip:$CALLEE$@192.168.7.51;transport=udp;user=phone>
CSeq: 1 INVITE
Contact: <sip:192.168.7.51:$SIP_PORT$>
Max-Forwards: 10
Content-Length: 154
Content-Type: application/sdp

v=0
o=alice 123456 789 IN IP4 192.168.7.51
s=SBC call
c=IN IP4 192.168.7.51
t=0 0
m=audio $MEDIA_PORT$ RTP/AVP 0
a=rtpmap:8 PCMA/8000
a=rtpmap:0 PCMU/8000

其中，$CALLEE$为被叫号码、$CALLER$为主叫号码、$CALL_ID$为会话标识

$SIP_PORT$为本地信令端口、$MEDIA_PORT$ 为本地媒体端口

注意：Content-Length的值为body的字符长度，计算方法如下：

sdp="v=0
o=alice 123456 789 IN IP4 192.168.7.51
s=SBC call
c=IN IP4 192.168.7.51
t=0 0
m=audio 18628 RTP/AVP 0
a=rtpmap:8 PCMA/8000
a=rtpmap:0 PCMU/8000"

echo "$sdp" | awk '{ bytes += length($0) + 2 } END { print bytes }'

二、sipsak模拟压测脚本 

#!/bin/bah

while true;do
    #暂停3～10秒
    sleep $(shuf -i 3-10 -n 1)
    initPort=$(shuf -i 40000-60000 -n 1)
    #发起10个invite请求
    for ((i = 0; i < 10; i++)); do
        sPt=$((initPort + i * 2))
        mPt=$((sPt + 1))
        sipsak -f invite.txt -g '!CALLEE!567831!CALL_ID!'$(uuidgen)'!CALLER!+8617621!SIP_PORT!'$sPt'!MEDIA_PORT!'$mPt'!' -H 172.23.33.3 -l $sPt -s sip:172.23.33.8:18627 -v
    done
done
