#将本地仓库关联到名为"origin"的远程仓库
git remote add origin https://gitee.com/zoewang/repo.git

#从名为"origin"的远程仓库中获取最新的提交信息同步到本地仓库
git fetch origin

#将本地重置为服务器端的版本
git reset --hard origin/master

#添加当前目录下的所有文件到暂存区
git add .

#提交暂存区文件到本地仓库
git commit -m "[message]"

#修改文件后不需要执行 git add 命令，直接来提交
git commit -a

#将本地的分支版本上传到远程并合并
git push

#强制将本地的分支版本上传到远程并合并
git push -f origin master

#配置用户名
git config --global user.name "your name"

#配置邮箱
git config --global user.email "your email"
