Freeswitch 1.10.6的docker镜像构建

Dockerfile文件内容

FROM debian:10-slim
ENV TZ=Asia/Shanghai

RUN apt-get update
RUN apt-get install apt-transport-https ca-certificates -y
RUN echo "deb http://mirrors.tuna.tsinghua.edu.cn/debian/ buster main contrib non-free" > /etc/apt/sources.list
RUN echo "deb http://mirrors.tuna.tsinghua.edu.cn/debian/ buster-updates main contrib non-free" >> /etc/apt/sources.list
RUN echo "deb http://mirrors.tuna.tsinghua.edu.cn/debian/ buster-backports main contrib non-free" >> /etc/apt/sources.list
RUN echo "deb http://mirrors.tuna.tsinghua.edu.cn/debian-security buster/updates main contrib non-free" >> /etc/apt/sources.list
RUN apt-get update
RUN apt-get install gcc g++ autoconf automake make libtool libtool-bin pkg-config \
    sngrep uuid-dev libtiff-dev libsqlite3-dev libcurl4-openssl-dev \
    libpcre3-dev libspeex-dev libedit-dev libcurl4-openssl-dev libshout3-dev \
    libavformat-dev libswscale-dev \
    libspeexdsp-dev libssl-dev yasm liblua5.1-0-dev libopus-dev libmpg123-dev \
    libmp3lame-dev libsndfile1-dev git -y --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*

RUN cd /usr/local/ \
    && git clone https://gitee.com/zoewang/spandsp.git \
    && git clone https://gitee.com/zoewang/sofia-sip.git \
    && git clone https://gitee.com/zoewang/freeswitch.git freeswitch-1.10.6

RUN cd /usr/local/spandsp/ \
    && ./bootstrap.sh -j && ./configure && make && make install && ldconfig
RUN cd /usr/local/sofia-sip/ \
    && ./bootstrap.sh -j && ./configure.gnu && make && make install && ldconfig

COPY modules.conf /usr/local/freeswitch-1.10.6

WORKDIR /usr/local/freeswitch-1.10.6
RUN ./bootstrap.sh
RUN ./configure --disable-dependency-tracking
RUN make && make install
RUN rm -rf /etc/localtime \
    && ln -s /usr/share/zoneinfo/Asia/Shanghai /etc/localtime \
    && echo 'export PATH="/usr/local/freeswitch/bin:$PATH"' >> /root/.bashrc

ENTRYPOINT ["/usr/local/freeswitch/bin/freeswitch","-c","-nonat"]

# Makefile Version 1.2.0
# 路径映射
# /usr/local/freeswitch/conf
# /usr/local/freeswitch/sounds
# /usr/local/freeswitch/scripts
# /usr/local/freeswitch/log
# /usr/local/freeswitch/recordings

init:
        -docker rm -f fs;
        docker run -d --network host \
        --name fs freeswitch:1.10.6
copy-conf:
        docker cp fs:/usr/local/freeswitch/conf ./
run:
        -docker rm -f fs;
        docker run -d --network host \
        -v $$PWD/conf:/usr/local/freeswitch/conf \
        -v $$PWD/log:/usr/local/freeswitch/log \
        --name fs freeswitch:1.10.6
